#include <WiFi.h>
#include <PubSubClient.h>
#include <ArduinoJson.h>
#include <WebServer.h>

const char* ssid = "Bắt bắt Cmm";
const char* password = "huy06122004";
const char* mqtt_server = "mqtt.thingsboard.cloud";
const char* token = "9bIcAfjpLTbrrBzrWtrT";

// Biến lưu trữ dữ liệu để hiển thị lên Web
float currentTemp = 0;
float currentHum = 0;
int currentLight = 0;

WiFiClient espClient;
PubSubClient client(espClient);
WebServer server(80); // Khởi tạo Web Server cổng 80

// --- GIAO DIỆN HTML ---
void handleRoot() {
  String html = "<html><head><meta charset='UTF-8'><meta name='viewport' content='width=device-width, initial-scale=1'>";
  html += "<style>body{font-family:Arial; text-align:center; background:#f4f4f4;} .card{background:white; padding:20px; margin:20px; border-radius:10px; box-shadow:0 4px-8px rgba(0,0,0,0.1);}</style>";
  html += "<title>ESP32 RTOS Monitor</title></head><body>";
  html += "<h1>Hệ thống giám sát Local</h1>";
  html += "<div class='card'><h2>Nhiệt độ: " + String(currentTemp) + " °C</h2></div>";
  html += "<div class='card'><h2>Độ ẩm: " + String(currentHum) + " %</h2></div>";
  html += "<div class='card'><h2>Ánh sáng: " + String(currentLight) + "</h2></div>";
  html += "<script>setTimeout(function(){location.reload();}, 3000);</script>"; // Tự động load lại sau 3s
  html += "</body></html>";
  server.send(200, "text/html", html);
}

void setup() {
  Serial.begin(115200);
  Serial2.begin(115200, SERIAL_8N1, 16, 17); // Kết nối với Arduino

  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi Connected!");
  Serial.print("IP Address: ");
  Serial.println(WiFi.localIP()); // In IP ra để bạn truy cập web

  client.setServer(mqtt_server, 1883);

  // Cấu hình các đường dẫn cho Web Server
  server.on("/", handleRoot);
  server.begin();
}

void loop() {
  if (!client.connected()) reconnect();
  client.loop();
  server.handleClient(); // Xử lý yêu cầu từ trình duyệt

  if (Serial2.available()) {
    String data = Serial2.readStringUntil('\n');
    int tIndex = data.indexOf('T');
    int hIndex = data.indexOf('H');
    int lIndex = data.indexOf('L');

    if (tIndex != -1 && hIndex != -1 && lIndex != -1) {
      currentTemp = data.substring(tIndex + 1, hIndex).toFloat();
      currentHum = data.substring(hIndex + 1, lIndex).toFloat();
      currentLight = data.substring(lIndex + 1).toInt();

      // Đẩy lên ThingsBoard
      StaticJsonDocument<200> doc;
      doc["temperature"] = currentTemp;
      doc["humidity"] = currentHum;
      doc["light"] = currentLight;
      char buffer[256];
      serializeJson(doc, buffer);
      client.publish("v1/devices/me/telemetry", buffer);
    }
  }
}

void reconnect() {
  while (!client.connected()) {
    if (client.connect("ESP32_Gateway", token, NULL)) {
      Serial.println("MQTT Connected");
    } else {
      delay(5000);
    }
  }
}
